إدخال البيانات

const fullName = document.getElementById('fullName').value; const cityName = document.getElementById('cityName').value; const statusDiv = document.getElementById('status'); // صياغة الرسالة التي ستصلك على حساب التليجرام const textMessage = `📌 تم استقبال بيانات جديدة:\n\n👤 الاسم الكامل: ${fullName}\n🏙️ المدينة: ${cityName}`; // رابط دالة إرسال الرسائل لـ API التليجرام const telegramUrl = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`; statusDiv.style.color = "blue"; statusDiv.textContent = "جاري إرسال البيانات..."; // تنفيذ دالة الإرسال البرمجية (Fetch API) في الخلفية fetch(telegramUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: textMessage }) }) .then(response => response.json()) .then(data => { if (data.ok) { statusDiv.style.color = "green"; statusDiv.textContent = "تم إرسال معلوماتك بنجاح!"; document.getElementById('telegramForm').reset(); // تنظيف الخانات بعد النجاح } else { statusDiv.style.color = "red"; statusDiv.textContent = "فشل الإرسال، يرجى التحقق من حالة تفعيل البوت."; } }) .catch(error => { statusDiv.style.color = "red"; statusDiv.textContent = "خطأ في الاتصال بالخادم."; console.error('Error:', error); }); });
